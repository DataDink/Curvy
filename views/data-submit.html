data-submit

Binds a command handler on the view-model to the element's "submit" event. 

   <div view-model="my-vm">
      <form data-submit="search">
         <button type="submit">Submit</button>
      </form>
   </div>

   app.viewmodel('my-vm', function() {
      this.search = function(e) {
         e.preventDefault();
         console.log('search initiated');
      };
   });
