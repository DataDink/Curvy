data-click

Binds a command handler on the view-model to the element's "click" event. 

   <div view-model="my-vm">
      <button data-click="logClick">Click Me</button>
   </div>

   app.viewmodel('my-vm', function() {
      this.logClick = function(e) {
         e.preventDefault();
         console.log('Clicked!');
      };
   });
